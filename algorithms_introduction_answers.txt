Making coffee:

  FUNCTION makeCoffee 

    IF coffee is not ground

      CALL grindCoffeeBeans

    ELSE 

      INSERT filter into coffeeMachine

      ADD coffee

      ADD water

      SET brewCoffee to true

    END IF

  END FUNCTION


Washing dishes:

  FUNCTION washDishes

    WHILE dishes are dirty THEN

      ADD water to dishes

      ADD soap to dishes

      SCRUB dishes

      ADD water to dishes

      ADD dishes to dish rack

    END WHILE

    IF dishes are not dirty

      ADD dishes to dish rack

  END FUNCTION
